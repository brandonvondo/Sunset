Behind Sunset Hills

$(".inp").on("keydown", function (e) { /*simple input verification that allows numbers tab and backspace*/
return ((e.keyCode >= 48 && e.keyCode <= 57) || (e.keyCode >= 96 && e.keyCode <= 105) || e.keyCode == 8 || e.keyCode == 9);
});

$("#sshbtn").on("click", function () {

if ($("#inpk").val() != "") { /*check to see if the user left a field blank*/
var bheight = [ /*creating an array with our 8 input values*/
Number($("#inp1").val()),
Number($("#inp2").val()),
Number($("#inp3").val()),
Number($("#inp4").val()),
Number($("#inp5").val()),
Number($("#inp6").val()),
Number($("#inp7").val()),
Number($("#inp8").val())
]

var bhigh = bheight[0]; /*setting high var to the starting value of our array as the first building always sees the sun*/
var output = [bhigh]; /*putting our starting building into our output array and also defining our output array*/

for (let loop = 1; loop < bheight.length; loop++) { /*for loop to runthrough all of our building height inputs*/
if (bheight[loop] > bhigh) { /*checking if the input we are currently on is higher than our previous high*/
bhigh = bheight[loop]; /*if it is we take the current loop & give it to the variable bhigh*/
output.push(bhigh); /*lastly we push bhigh into our output array for later*/
};
}; /*here we print how many/ which buildings can see the sun*/
$("#sshout").html(`There are ${output.length} building(s) that can see the sun, and those are as follows: ${output}.`)
}
else $("#sshout").html(`Please enter real numbers.`) /*input validation for blank spaces*/
});

$("#btnclear").on("click", function () { /*simple button to clear the results and the input field*/
$(".inp").val("");
$("#sshout").text("");
});
